const nuevoProyecto = {
    nombre: 'Proyecto 1',

}

// Función para crear un nuevo objetivo
function crearObjetivo() {
    // Obtener los valores de los campos de entrada
    const nombre = document.getElementById("nombre").value;
    const fechaInicio = document.getElementById("fecha-1").value;

    // Generar un nuevo ID para el objetivo
    const lasId = 1;
    nuevoProyecto.objetivos[lasId + 1] = {
        nombre: nombre,
        fechaInicio: fechaInicio,
    }

    // Crear un nuevo elemento div para mostrar el objetivo
    const nuevoObjetivo = document.createElement('div');
    nuevoObjetivo.innerHTML = `
    <div class="border rounded p-3 my-3 objetivo">
        <p>Nombre Objetivo: ${nombre}</p>
        <p>Fecha Inicio: ${fechaInicio}</p>
    </div>`;

    // Crear un botón para eliminar el objetivo
    const button = document.createElement('button');
    button.textContent = 'Eliminar';
    button.onclick = () => nuevoObjetivo.remove();
    nuevoObjetivo.appendChild(button);

    // Agregar el nuevo objetivo al contenedor de objetivos
    const contenedorObjetivos = document.getElementById("contenedor-objetivos");
    contenedorObjetivos.appendChild(nuevoObjetivo);
}

// Obtener el botón de "Nuevo" y asignarle la función crearObjetivo como evento onclick
const button = document.querySelector('#new');
button.onclick = () => crearObjetivo();
